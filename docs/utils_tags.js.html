/**
 * Merges two tag sets.
 * @memberof Utils
 * @param {Tag[] | undefined} existingTags
 * @param {Tags} newTags
 * @returns {Tag[]}
 */
const mergeTags = (existingTags, newTags) => {
    const newTagSet = Object.keys(newTags).map((key) => ({
        Key: key,
        Value: newTags[key],
    }));
    const mergedTagSet = [...(existingTags ?? [])];
    newTagSet.forEach((newTag) => {
        const existingTagIndex = mergedTagSet.findIndex((tag) => tag.Key === newTag.Key);
        if (existingTagIndex > -1) {
            mergedTagSet[existingTagIndex].Value = newTag.Value;
        }
        else {
            mergedTagSet.push(newTag);
        }
    });
    return mergedTagSet;
};
exports.mergeTags = mergeTags;
